using Polynomials
using Plots

function Legendre(n)
    if(n == 0)
        return Polynomial([1])
    elseif(n == 1)
        return Polynomial([0, 1])
    else
        return (2n-1)/(n)*Polynomial([0,1])*Legendre(n-1) - (n-1)/(n)*Legendre(n-2)
    end
    
end

plot(Legendre(4),-1,1, label="Legendre(4)", xlabel="x", ylabel="f(x)", title="wielomiany Legendra")
plot!(Legendre(3),-1,1, label="Legendre(3)")
plot!(Legendre(2),-1,1, label="Legendre(2)")
plot!(Legendre(1),-1,1, label="Legendre(1)")

function check_indirect(k)
    i = 0
    while(true)
        f(x)=x^i 
        result = "rwne"
        integ, err = quadgk(f, -1, 1)
        if(!isapprox(integ, integral(f, k), atol = 1e-14))
            print("niedokadny dla stopnia: ", i,'\n')
            return
        end
        i = i + 1
    end
end

function integral2(f,a,b,k)
    dx = (b-a)/k
    current = a
    sum = 0
    for i in 1:(k-1)
        sum += (f(current) + f(current+dx))*dx/2
        current += dx
    end
    return sum
end
