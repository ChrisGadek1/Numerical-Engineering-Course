for(i in 1:length(covid_data_poland$new_cases)){
  if(i > 7){
    sum = 0
    for(j in (i-6):i){
      sum = sum + covid_data_poland$new_cases[j]
    }
    sum = sum / 6
    avg_7_day <- append(avg_7_day, sum)
